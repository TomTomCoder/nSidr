## Task 1 — Backend: POST /plugin/:id/consent

In `plugin.service.ts`, add:
```typescript
async consentExtension(pluginId: string, spaceId: string): Promise<void> {
  const plugin = await this.prismaService.plugin.findUniqueOrThrow({ where: { id: pluginId } });
  if (!plugin.isExtension) throw new CustomHttpException(VALIDATION_ERROR, 'Not an extension plugin');
  await this.prismaService.plugin.update({
    where: { id: pluginId },
    data: { consentedAt: new Date() },
  });
}

async revokeConsent(pluginId: string): Promise<void> {
  await this.prismaService.plugin.update({
    where: { id: pluginId, isExtension: true },
    data: { consentedAt: null },
  });
}
```

In `plugin.controller.ts`, add:
```typescript
@Post(':id/consent')
@Permissions('space|create')
async consentExtension(@Param('id') id: string, @Body() body: { spaceId: string }) {
  return this.pluginService.consentExtension(id, body.spaceId);
}

@Delete(':id/consent')
@Permissions('space|create')
async revokeConsent(@Param('id') id: string) {
  return this.pluginService.revokeConsent(id);
}
```
