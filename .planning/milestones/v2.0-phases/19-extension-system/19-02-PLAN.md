## Task 2 — installByUrl() method in PluginService

In `apps/nestjs-backend/src/features/plugin/plugin.service.ts`:

Add method:
```typescript
async installByUrl(spaceId: string, mcpUrl: string, installedBy: string): Promise<{ pluginId: string }> {
  // 1. SSRF guard (T-19-01)
  await this.ssrfGuardService.assertSafe(mcpUrl);

  // 2. Fetch MCP manifest via tools/list
  const manifest = await this.fetchMcpManifest(mcpUrl); // see below

  // 3. Register Plugin record (isExtension=true, status='published')
  const secret = generateSecret(); // reuse existing pattern
  const plugin = await this.prismaService.$tx(async (prisma) => {
    const record = await prisma.plugin.create({
      data: {
        name: manifest.name ?? new URL(mcpUrl).hostname,
        logo: '',
        status: 'published',
        positions: '[]',
        secret,
        maskedSecret: maskSecret(secret),
        mcpUrl,
        toolManifest: manifest.tools, // whitelist: [{name, description, inputSchema}]
        isExtension: true,
        requestedScopes: manifest.tools.map(t => t.name),
        consentedAt: null, // not yet consented
        createdBy: installedBy,
      },
    });
    // 4. Install into space (position='extension')
    await prisma.pluginInstall.create({
      data: {
        pluginId: record.id,
        baseId: spaceId,
        name: record.name,
        positionId: spaceId,
        position: 'extension',
        createdBy: installedBy,
      },
    });
    return record;
  });
  return { pluginId: plugin.id };
}

private async fetchMcpManifest(mcpUrl: string): Promise<{ name?: string; tools: ManifestTool[] }> {
  let res: Response;
  try {
    res = await fetch(mcpUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ jsonrpc: '2.0', id: 1, method: 'tools/list', params: {} }),
      signal: AbortSignal.timeout(5000),
    });
  } catch (e) {
    throw new CustomHttpException(VALIDATION_ERROR, `Cannot reach MCP server at ${mcpUrl}: ${e.message}`);
  }
  if (!res.ok) throw new CustomHttpException(VALIDATION_ERROR, `MCP server returned HTTP ${res.status}`);
  const body = await res.json() as { result?: { tools?: unknown[] } };
  const rawTools = body?.result?.tools ?? [];
  // T-19-02: whitelist fields only
  const tools = (Array.isArray(rawTools) ? rawTools : []).map((t: Record<string, unknown>) => ({
    name: String(t.name ?? ''),
    description: t.description ? String(t.description) : undefined,
    inputSchema: (t.inputSchema && typeof t.inputSchema === 'object') ? t.inputSchema : { type: 'object', properties: {} },
  })).filter(t => t.name);
  return { name: undefined, tools };
}
```
