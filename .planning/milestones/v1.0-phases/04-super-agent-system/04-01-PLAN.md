```prisma
model Agent {
  id               String    @id @default(cuid())
  name             String
  description      String?
  baseId           String    @map("base_id")
  instructions     String?   @db.Text
  modelKey         String?   @map("model_key")
  isPublic         Boolean   @default(false) @map("is_public")
  isActive         Boolean   @default(true) @map("is_active")
  maxIterations    Int       @default(10) @map("max_iterations")
  createdBy        String    @map("created_by")
  createdTime      DateTime  @default(now()) @map("created_time")
  lastModifiedTime DateTime? @updatedAt @map("last_modified_time")
  lastModifiedBy   String?   @map("last_modified_by")

  tools       AgentTool[]
  triggers    AgentTrigger[]
  memories    AgentMemory[]
  connections AgentConnection[]

  @@index([baseId])
  @@index([createdBy])
  @@map("agent")
}

model AgentTool {
  id        String  @id @default(cuid())
  agentId   String  @map("agent_id")
  toolName  String  @map("tool_name")
  isEnabled Boolean @default(true) @map("is_enabled")

  agent Agent @relation(fields: [agentId], references: [id], onDelete: Cascade)

  @@unique([agentId, toolName])
  @@index([agentId])
  @@map("agent_tool")
}

model AgentTrigger {
  id          String   @id @default(cuid())
  agentId     String   @map("agent_id")
  triggerType String   @map("trigger_type")
  config      Json?
  isActive    Boolean  @default(true) @map("is_active")
  createdTime DateTime @default(now()) @map("created_time")

  agent Agent @relation(fields: [agentId], references: [id], onDelete: Cascade)

  @@index([agentId])
  @@map("agent_trigger")
}

model AgentMemory {
  id          String    @id @default(cuid())
  agentId     String    @map("agent_id")
  memoryType  String    @map("memory_type")
  content     String    @db.Text
  metadata    Json?
  expiresAt   DateTime? @map("expires_at")
  createdTime DateTime  @default(now()) @map("created_time")

  agent Agent @relation(fields: [agentId], references: [id], onDelete: Cascade)

  @@index([agentId, memoryType])
  @@index([expiresAt])
  @@map("agent_memory")
}

model AgentConnection {
  id             String   @id @default(cuid())
  agentId        String   @map("agent_id")
  provider       String
  encryptedToken String?  @map("encrypted_token") @db.Text
  scopes         String[]
  isEnabled      Boolean  @default(false) @map("is_enabled")
  createdTime    DateTime @default(now()) @map("created_time")
  lastModifiedTime DateTime? @updatedAt @map("last_modified_time")

  agent Agent @relation(fields: [agentId], references: [id], onDelete: Cascade)

  @@unique([agentId, provider])
  @@index([agentId])
  @@map("agent_connection")
}
```
