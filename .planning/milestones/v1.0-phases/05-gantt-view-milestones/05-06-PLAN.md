    ```typescript
    import type { IGanttViewOptions } from '@teable/core';
    import { BadRequestException } from '@nestjs/common';
    import type { PrismaService } from '@teable/db-main-prisma';

    export async function validateGanttViewOptions(
      tableId: string,
      options: IGanttViewOptions,
      prisma: PrismaService
    ): Promise<void> {
      // Fetch all fields for this table
      const fields = await prisma.field.findMany({
        where: { tableId },
        select: { id: true, type: true },
      });

      const fieldMap = new Map(fields.map((f) => [f.id, f.type]));

      // Validate startField: must exist and be date or dateTime type
      const startType = fieldMap.get(options.startField);
      if (!startType) {
        throw new BadRequestException('startField references a non-existent field');
      }
      if (startType !== 'date' && startType !== 'dateTime') {
        throw new BadRequestException('startField must reference a date or dateTime field');
      }

      // Validate endField: must exist and be date or dateTime type
      const endType = fieldMap.get(options.endField);
      if (!endType) {
        throw new BadRequestException('endField references a non-existent field');
      }
      if (endType !== 'date' && endType !== 'dateTime') {
        throw new BadRequestException('endField must reference a date or dateTime field');
      }

      // Validate optional fields: existence check only (any type)
      if (options.titleField) {
        if (!fieldMap.has(options.titleField)) {
          throw new BadRequestException('titleField references a non-existent field');
        }
      }
      if (options.dependencyField) {
        if (!fieldMap.has(options.dependencyField)) {
          throw new BadRequestException('dependencyField references a non-existent field');
        }
      }
      if (options.colorField) {
        if (!fieldMap.has(options.colorField)) {
          throw new BadRequestException('colorField references a non-existent field');
        }
      }
    }
    ```
