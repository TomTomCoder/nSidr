```typescript
import { Injectable } from '@nestjs/common';
import { PrismaService } from '../../db-provider/prisma.service';
import { EmbeddingService } from './embedding.service';
import { Prisma } from '@teable/db-main-prisma';

export interface DocSearchResult {
  chunkId: string;
  docId: string;
  docTitle: string;
  chunkContent: string;
  score: number;
}

@Injectable()
export class DocSearchService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly embeddingService: EmbeddingService,
  ) {}

  async semanticSearch(spaceId: string, query: string, limit = 10): Promise<DocSearchResult[]> {
    const embedding = await this.embeddingService.generateEmbedding(query);
    const vectorStr = `[${embedding.join(',')}]`;

    const rows = await this.prisma.$queryRaw<
      Array<{ id: string; content: string; doc_id: string; title: string; distance: number }>
    >`
      SELECT dc.id, dc.content, dc."docId" AS doc_id, d.title,
             dc.embedding <=> ${vectorStr}::vector AS distance
      FROM "DocChunk" dc
      JOIN "ImportedDoc" d ON dc."docId" = d.id
      WHERE d."spaceId" = ${spaceId}
        AND d."isIndexed" = true
      ORDER BY dc.embedding <=> ${vectorStr}::vector
      LIMIT ${limit}
    `;

    return rows.map((r) => ({
      chunkId: r.id,
      docId: r.doc_id,
      docTitle: r.title,
      chunkContent: r.content,
      score: 1 - r.distance,
    }));
  }

  async keywordSearch(spaceId: string, query: string, limit = 10): Promise<DocSearchResult[]> {
    const rows = await this.prisma.$queryRaw<
      Array<{ id: string; content: string; doc_id: string; title: string; rank: number }>
    >`
      SELECT dc.id, dc.content, dc."docId" AS doc_id, d.title,
             ts_rank(to_tsvector('english', dc.content), plainto_tsquery('english', ${query})) AS rank
      FROM "DocChunk" dc
      JOIN "ImportedDoc" d ON dc."docId" = d.id
      WHERE d."spaceId" = ${spaceId}
        AND d."isIndexed" = true
        AND to_tsvector('english', dc.content) @@ plainto_tsquery('english', ${query})
      ORDER BY rank DESC
      LIMIT ${limit}
    `;

    return rows.map((r) => ({
      chunkId: r.id,
      docId: r.doc_id,
      docTitle: r.title,
      chunkContent: r.content,
      score: r.rank,
    }));
  }

  async hybridSearch(spaceId: string, query: string, limit = 10): Promise<DocSearchResult[]> {
    const [semantic, keyword] = await Promise.all([
      this.semanticSearch(spaceId, query, limit),
      this.keywordSearch(spaceId, query, limit),
    ]);

    const scoreMap = new Map<string, DocSearchResult>();

    for (const r of semantic) {
      scoreMap.set(r.chunkId, { ...r, score: 0.7 * r.score });
    }
    for (const r of keyword) {
      const existing = scoreMap.get(r.chunkId);
      if (existing) {
        existing.score += 0.3 * r.score;
      } else {
        scoreMap.set(r.chunkId, { ...r, score: 0.3 * r.score });
      }
    }

    return [...scoreMap.values()]
      .sort((a, b) => b.score - a.score)
      .slice(0, limit);
  }
}
```

```typescript
import { Injectable } from '@nestjs/common';
import { PrismaService } from '../../db-provider/prisma.service';

@Injectable()
export class DocGraphService {
  constructor(private readonly prisma: PrismaService) {}

  async getLinkedDocs(docId: string) {
    const [outbound, inbound] = await Promise.all([
      this.prisma.docLink.findMany({ where: { fromDocId: docId }, include: { toDoc: { select: { id: true, title: true } } } }),
      this.prisma.docLink.findMany({ where: { toDocId: docId }, include: { fromDoc: { select: { id: true, title: true } } } }),
    ]);
    return { outbound, inbound };
  }

  async getDocGraph(spaceId: string) {
    const [nodes, edges] = await Promise.all([
      this.prisma.importedDoc.findMany({ where: { spaceId }, select: { id: true, title: true, sourceType: true, isIndexed: true, chunkCount: true, wordCount: true } }),
      this.prisma.docLink.findMany({ where: { fromDoc: { spaceId } } }),
    ]);
    return { nodes, edges };
  }
}
```

```typescript
import { Controller, Post, Get, Delete, Param, Body, Query, UseGuards } from '@nestjs/common';
import { DocSearchService } from './search.service';
import { DocGraphService } from './graph.service';
import { PrismaService } from '../../db-provider/prisma.service';
import { JwtAuthGuard } from '../../auth/guard/jwt.auth.guard';

@UseGuards(JwtAuthGuard)
@Controller('api/spaces/:spaceId/docs')
export class DocSearchController {
  constructor(
    private readonly searchService: DocSearchService,
    private readonly graphService: DocGraphService,
    private readonly prisma: PrismaService,
  ) {}

  @Post('search')
  async search(
    @Param('spaceId') spaceId: string,
    @Body() body: { query: string; mode?: 'semantic' | 'keyword' | 'hybrid'; limit?: number },
  ) {
    const { query, mode = 'hybrid', limit = 10 } = body;
    switch (mode) {
      case 'semantic':
        return this.searchService.semanticSearch(spaceId, query, limit);
      case 'keyword':
        return this.searchService.keywordSearch(spaceId, query, limit);
      default:
        return this.searchService.hybridSearch(spaceId, query, limit);
    }
  }

  @Get(':docId/links')
  async getLinks(@Param('docId') docId: string) {
    return this.graphService.getLinkedDocs(docId);
  }

  @Get(':docId')
  async getDoc(@Param('docId') docId: string) {
    const doc = await this.prisma.importedDoc.findUniqueOrThrow({
      where: { id: docId },
      include: {
        _count: { select: { chunks: true, linksFrom: true, linksTo: true } },
      },
    });
    return doc;
  }

  @Get()
  async listDocs(@Param('spaceId') spaceId: string) {
    return this.prisma.importedDoc.findMany({
      where: { spaceId },
      orderBy: { createdAt: 'desc' },
      select: { id: true, title: true, sourceType: true, wordCount: true, chunkCount: true, isIndexed: true, createdAt: true },
    });
  }

  @Delete(':docId')
  async deleteDoc(@Param('docId') docId: string) {
    // Cascade deletes DocChunk and DocLink via Prisma onDelete: Cascade
    await this.prisma.importedDoc.delete({ where: { id: docId } });
    return { deleted: true };
  }
}
```

## Notes

- DocIngestionService must cap chunk content at 2000 tokens before sending to EmbeddingService. If a chunk exceeds this limit, truncate it and log a warning. This limits vector injection surface (see T-07-03-06).
- The `<=>` operator is pgvector's cosine distance operator (0 = identical, 2 = opposite). `score = 1 - distance` converts to similarity.
- For keyword search, `ts_rank` returns a float in [0, 1]. Hybrid weighting assumes both scores are normalized — this is an approximation; a future phase can normalize properly.
- The `getDocGraph` endpoint is intended for visualization in Plan 04; it returns all nodes and edges and may be large for spaces with many documents. Pagination can be added in a future phase.
