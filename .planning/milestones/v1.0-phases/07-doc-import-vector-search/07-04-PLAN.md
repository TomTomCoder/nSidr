```typescript
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
import type { IDocSearchQuery, IDocSearchResult, IImportedDoc } from '@teable/openapi';

const docKeys = {
  list: (spaceId: string) => ['docs', spaceId, 'list'] as const,
  links: (docId: string) => ['docs', docId, 'links'] as const,
  doc: (docId: string) => ['docs', docId] as const,
};

export function useDocList(spaceId: string) {
  return useQuery({
    queryKey: docKeys.list(spaceId),
    queryFn: () =>
      fetch(`/api/spaces/${spaceId}/docs`).then((r) => r.json() as Promise<IImportedDoc[]>),
  });
}

export function useDocSearch(spaceId: string) {
  return useMutation<IDocSearchResult[], Error, IDocSearchQuery>({
    mutationFn: (body) =>
      fetch(`/api/spaces/${spaceId}/docs/search`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body),
      }).then((r) => r.json()),
  });
}

export function useImportMarkdown(spaceId: string) {
  const qc = useQueryClient();
  return useMutation<{ jobId: string; status: string }, Error, { title: string; content: string; userId: string }>({
    mutationFn: (body) =>
      fetch(`/api/spaces/${spaceId}/docs/import/markdown`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(body),
      }).then((r) => r.json()),
    onSuccess: () => qc.invalidateQueries({ queryKey: docKeys.list(spaceId) }),
  });
}

export function useImportPdf(spaceId: string) {
  const qc = useQueryClient();
  return useMutation<{ jobId: string; status: string }, Error, { title: string; file: File; userId: string }>({
    mutationFn: ({ title, file, userId }) => {
      const form = new FormData();
      form.append('file', file);
      form.append('title', title);
      form.append('userId', userId);
      return fetch(`/api/spaces/${spaceId}/docs/import/pdf`, { method: 'POST', body: form }).then((r) => r.json());
    },
    onSuccess: () => qc.invalidateQueries({ queryKey: docKeys.list(spaceId) }),
  });
}

export function useDocLinks(spaceId: string, docId: string) {
  return useQuery({
    queryKey: docKeys.links(docId),
    queryFn: () => fetch(`/api/spaces/${spaceId}/docs/${docId}/links`).then((r) => r.json()),
    enabled: !!docId,
  });
}

export function useDoc(spaceId: string, docId: string) {
  return useQuery({
    queryKey: docKeys.doc(docId),
    queryFn: () =>
      fetch(`/api/spaces/${spaceId}/docs/${docId}`).then((r) => r.json() as Promise<IImportedDoc & { rawContent: string }>),
    enabled: !!docId,
  });
}

export function useDeleteDoc(spaceId: string) {
  const qc = useQueryClient();
  return useMutation<{ deleted: boolean }, Error, string>({
    mutationFn: (docId) =>
      fetch(`/api/spaces/${spaceId}/docs/${docId}`, { method: 'DELETE' }).then((r) => r.json()),
    onSuccess: () => qc.invalidateQueries({ queryKey: docKeys.list(spaceId) }),
  });
}
```

```tsx
'use client';
import { useState, useEffect } from 'react';
import type { IDocSearchResult } from '@teable/openapi';
import { useDocSearch } from './hooks';

interface DocSearchPanelProps {
  spaceId: string;
  open: boolean;
  onClose: () => void;
  onSelectResult: (result: IDocSearchResult) => void;
}

type SearchMode = 'semantic' | 'keyword' | 'hybrid';

export function DocSearchPanel({ spaceId, open, onClose, onSelectResult }: DocSearchPanelProps) {
  const [query, setQuery] = useState('');
  const [mode, setMode] = useState<SearchMode>('hybrid');
  const { mutate: search, data: results, isPending } = useDocSearch(spaceId);

  useEffect(() => {
    if (!open) return;
    const handler = (e: KeyboardEvent) => {
      if (e.key === 'Escape') onClose();
    };
    window.addEventListener('keydown', handler);
    return () => window.removeEventListener('keydown', handler);
  }, [open, onClose]);

  if (!open) return null;

  const handleSearch = (q: string) => {
    setQuery(q);
    if (q.trim().length > 1) search({ query: q, mode, limit: 10 });
  };

  return (
    <div className="fixed inset-0 z-50 flex items-start justify-center pt-20" onClick={onClose}>
      <div
        className="bg-white dark:bg-zinc-900 rounded-xl shadow-2xl w-full max-w-2xl p-4"
        onClick={(e) => e.stopPropagation()}
      >
        <div className="flex gap-2 mb-3">
          {(['semantic', 'keyword', 'hybrid'] as SearchMode[]).map((m) => (
            <button
              key={m}
              onClick={() => setMode(m)}
              className={`px-3 py-1 rounded text-sm capitalize ${mode === m ? 'bg-blue-600 text-white' : 'bg-zinc-100 dark:bg-zinc-800 text-zinc-600 dark:text-zinc-300'}`}
            >
              {m}
            </button>
          ))}
        </div>
        <input
          autoFocus
          className="w-full border rounded-lg px-3 py-2 text-sm outline-none focus:ring-2 focus:ring-blue-500 dark:bg-zinc-800 dark:border-zinc-700 dark:text-white"
          placeholder="Search documents..."
          value={query}
          onChange={(e) => handleSearch(e.target.value)}
        />
        {isPending && <p className="text-xs text-zinc-400 mt-2">Searching...</p>}
        <ul className="mt-3 max-h-96 overflow-y-auto space-y-2">
          {results?.map((r) => (
            <li
              key={r.chunkId}
              onClick={() => onSelectResult(r)}
              className="cursor-pointer p-3 rounded-lg hover:bg-zinc-100 dark:hover:bg-zinc-800"
            >
              <div className="flex justify-between items-center">
                <span className="font-medium text-sm text-zinc-800 dark:text-zinc-100">{r.docTitle}</span>
                <span className="text-xs text-zinc-400">{(r.score * 100).toFixed(0)}%</span>
              </div>
              {/* Render as plain text — document content is untrusted, never render as HTML */}
              <p className="text-xs text-zinc-500 mt-1 line-clamp-2">{r.chunkContent}</p>
            </li>
          ))}
        </ul>
      </div>
    </div>
  );
}
```

```tsx
'use client';
import { useState, useRef } from 'react';
import { useImportMarkdown, useImportPdf } from './hooks';

interface DocImportPanelProps {
  spaceId: string;
  userId: string;
  onClose: () => void;
}

export function DocImportPanel({ spaceId, userId, onClose }: DocImportPanelProps) {
  const [tab, setTab] = useState<'markdown' | 'pdf'>('markdown');
  const [title, setTitle] = useState('');
  const [content, setContent] = useState('');
  const [file, setFile] = useState<File | null>(null);
  const [jobId, setJobId] = useState<string | null>(null);
  const fileRef = useRef<HTMLInputElement>(null);

  const { mutate: importMd, isPending: mdPending } = useImportMarkdown(spaceId);
  const { mutate: importPdf, isPending: pdfPending } = useImportPdf(spaceId);

  const isPending = mdPending || pdfPending;

  const handleImport = () => {
    if (tab === 'markdown') {
      importMd({ title, content, userId }, { onSuccess: (d) => setJobId(d.jobId) });
    } else if (file) {
      // Validate PDF MIME type on the client before uploading
      if (file.type !== 'application/pdf') {
        alert('Only PDF files are accepted');
        return;
      }
      // Cap at 50 MB on the client
      if (file.size > 50 * 1024 * 1024) {
        alert('File exceeds 50 MB limit');
        return;
      }
      importPdf({ title, file, userId }, { onSuccess: (d) => setJobId(d.jobId) });
    }
  };

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center" onClick={onClose}>
      <div
        className="bg-white dark:bg-zinc-900 rounded-xl shadow-2xl w-full max-w-lg p-5"
        onClick={(e) => e.stopPropagation()}
      >
        <h2 className="text-lg font-semibold mb-4 dark:text-white">Import Document</h2>
        <div className="flex gap-2 mb-4">
          {(['markdown', 'pdf'] as const).map((t) => (
            <button
              key={t}
              onClick={() => setTab(t)}
              className={`px-4 py-1.5 rounded text-sm uppercase tracking-wide ${tab === t ? 'bg-blue-600 text-white' : 'bg-zinc-100 dark:bg-zinc-800 text-zinc-600 dark:text-zinc-300'}`}
            >
              {t}
            </button>
          ))}
        </div>
        <input
          className="w-full border rounded-lg px-3 py-2 text-sm mb-3 outline-none focus:ring-2 focus:ring-blue-500 dark:bg-zinc-800 dark:border-zinc-700 dark:text-white"
          placeholder="Document title"
          value={title}
          onChange={(e) => setTitle(e.target.value)}
        />
        {tab === 'markdown' ? (
          <textarea
            className="w-full border rounded-lg px-3 py-2 text-sm h-40 outline-none focus:ring-2 focus:ring-blue-500 dark:bg-zinc-800 dark:border-zinc-700 dark:text-white font-mono resize-none"
            placeholder="Paste markdown content here..."
            value={content}
            onChange={(e) => setContent(e.target.value)}
          />
        ) : (
          <div
            className="border-2 border-dashed rounded-lg p-8 text-center cursor-pointer hover:border-blue-400 dark:border-zinc-700"
            onClick={() => fileRef.current?.click()}
            onDrop={(e) => { e.preventDefault(); setFile(e.dataTransfer.files[0] ?? null); }}
            onDragOver={(e) => e.preventDefault()}
          >
            {file ? (
              <p className="text-sm text-zinc-600 dark:text-zinc-300">{file.name}</p>
            ) : (
              <p className="text-sm text-zinc-400">Drag and drop a PDF (max 50 MB), or click to browse</p>
            )}
            <input ref={fileRef} type="file" accept="application/pdf" className="hidden" onChange={(e) => setFile(e.target.files?.[0] ?? null)} />
          </div>
        )}
        {jobId && (
          <div className="mt-3 p-2 bg-green-50 dark:bg-green-900/30 rounded text-xs text-green-700 dark:text-green-300">
            Queued (job {jobId}) — processing in background. The document will appear in the library once indexed.
          </div>
        )}
        <div className="flex justify-end gap-2 mt-4">
          <button onClick={onClose} className="px-4 py-2 text-sm rounded-lg bg-zinc-100 dark:bg-zinc-800 dark:text-zinc-300">Cancel</button>
          <button
            onClick={handleImport}
            disabled={isPending || !title || (tab === 'markdown' ? !content : !file)}
            className="px-4 py-2 text-sm rounded-lg bg-blue-600 text-white disabled:opacity-50"
          >
            {isPending ? 'Queueing...' : 'Import'}
          </button>
        </div>
      </div>
    </div>
  );
}
```

```tsx
'use client';
import type { IImportedDoc } from '@teable/openapi';
import { useDoc, useDocLinks } from './hooks';

interface DocViewerProps {
  doc: IImportedDoc;
  spaceId: string;
  onClose: () => void;
}

export function DocViewer({ doc, spaceId, onClose }: DocViewerProps) {
  // Fetch full doc record (including rawContent) — the list endpoint omits rawContent
  const { data: fullDoc } = useDoc(spaceId, doc.id);
  const { data: links } = useDocLinks(spaceId, doc.id);

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/40" onClick={onClose}>
      <div
        className="bg-white dark:bg-zinc-900 rounded-xl shadow-2xl w-full max-w-3xl max-h-[80vh] flex flex-col"
        onClick={(e) => e.stopPropagation()}
      >
        <div className="flex items-center justify-between px-6 py-4 border-b dark:border-zinc-700">
          <div>
            <h2 className="text-lg font-semibold dark:text-white">{doc.title}</h2>
            <p className="text-xs text-zinc-400 mt-0.5">
              {doc.wordCount.toLocaleString()} words · {doc.chunkCount} chunks · {doc.sourceType}
            </p>
          </div>
          <button onClick={onClose} className="text-zinc-400 hover:text-zinc-600 text-xl leading-none">&times;</button>
        </div>
        <div className="overflow-y-auto flex-1 px-6 py-4">
          {/* Plain text rendering only — document content is untrusted and must never be rendered as HTML */}
          <pre className="text-sm text-zinc-700 dark:text-zinc-300 whitespace-pre-wrap font-sans break-words">
            {fullDoc?.rawContent ?? 'Loading…'}
          </pre>
        </div>
        {links && (links.outbound.length > 0 || links.inbound.length > 0) && (
          <div className="px-6 py-4 border-t dark:border-zinc-700">
            {links.outbound.length > 0 && (
              <div className="mb-2">
                <p className="text-xs font-semibold text-zinc-500 uppercase tracking-wide mb-1">Links to</p>
                <ul className="flex flex-wrap gap-1">
                  {links.outbound.map((l) => (
                    <li key={l.id} className="text-xs bg-blue-50 dark:bg-blue-900/30 text-blue-600 dark:text-blue-300 px-2 py-0.5 rounded">
                      {l.linkText}
                    </li>
                  ))}
                </ul>
              </div>
            )}
            {links.inbound.length > 0 && (
              <div>
                <p className="text-xs font-semibold text-zinc-500 uppercase tracking-wide mb-1">Linked from</p>
                <ul className="flex flex-wrap gap-1">
                  {links.inbound.map((l) => (
                    <li key={l.id} className="text-xs bg-purple-50 dark:bg-purple-900/30 text-purple-600 dark:text-purple-300 px-2 py-0.5 rounded">
                      {l.linkText}
                    </li>
                  ))}
                </ul>
              </div>
            )}
          </div>
        )}
      </div>
    </div>
  );
}
```

```tsx
'use client';
import { useState } from 'react';
import type { IImportedDoc } from '@teable/openapi';
import { useDocList, useDeleteDoc } from './hooks';
import { DocViewer } from './DocViewer';

interface DocLibraryProps {
  spaceId: string;
}

export function DocLibrary({ spaceId }: DocLibraryProps) {
  const { data: docs, isLoading } = useDocList(spaceId);
  const { mutate: deleteDoc } = useDeleteDoc(spaceId);
  const [selectedDoc, setSelectedDoc] = useState<IImportedDoc | null>(null);

  if (isLoading) return <div className="p-4 text-sm text-zinc-400">Loading documents...</div>;
  if (!docs?.length) return <div className="p-4 text-sm text-zinc-400">No documents imported yet.</div>;

  return (
    <div className="p-4">
      <h3 className="text-sm font-semibold text-zinc-500 uppercase tracking-wide mb-3">Document Library</h3>
      <ul className="space-y-2">
        {docs.map((doc) => (
          <li
            key={doc.id}
            className="flex items-center justify-between p-3 rounded-lg bg-zinc-50 dark:bg-zinc-800 hover:bg-zinc-100 dark:hover:bg-zinc-700 cursor-pointer"
            onClick={() => setSelectedDoc(doc as IImportedDoc)}
          >
            <div>
              <p className="text-sm font-medium dark:text-white">{doc.title}</p>
              <p className="text-xs text-zinc-400 mt-0.5">
                {doc.sourceType} · {doc.wordCount?.toLocaleString()} words · {doc.chunkCount} chunks ·{' '}
                <span className={doc.isIndexed ? 'text-green-500' : 'text-yellow-500'}>
                  {doc.isIndexed ? 'indexed' : 'indexing...'}
                </span>
              </p>
            </div>
            <button
              onClick={(e) => { e.stopPropagation(); deleteDoc(doc.id); }}
              className="ml-4 text-xs text-red-400 hover:text-red-600"
            >
              Delete
            </button>
          </li>
        ))}
      </ul>
      {selectedDoc && <DocViewer doc={selectedDoc} spaceId={spaceId} onClose={() => setSelectedDoc(null)} />}
    </div>
  );
}
```

| T-07-04-03 | Denial of Service | Large PDF uploaded via drag-and-drop | mitigate | Client-side check: `file.size > 50MB` shows error before upload; MIME type validated as `application/pdf`; server-side Multer limit also enforced (Plan 02) |
| T-07-04-04 | Information Disclosure | Search results expose doc content from other spaces | mitigate | Backend filters by `spaceId`; frontend always passes current space's `spaceId` from auth context |
