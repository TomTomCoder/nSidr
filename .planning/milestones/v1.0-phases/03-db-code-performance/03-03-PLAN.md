   ```typescript
   import { BullModule } from '@nestjs/bullmq';
   import { Module } from '@nestjs/common';
   import { BullBoardModule } from '@bull-board/nestjs';
   import { BullMQAdapter } from '@bull-board/api/bullMQAdapter';
   import { ExpressAdapter } from '@bull-board/express';
   import { IMPORT_QUEUE, AI_GENERATION_QUEUE } from './queue.types';
   import { ImportProcessor } from './processors/import.processor';
   import { AiGenerationProcessor } from './processors/ai-generation.processor';

   const defaultJobOptions = {
     attempts: 3,
     backoff: { type: 'exponential' as const, delay: 2000 },
     removeOnComplete: 100,
     removeOnFail: 500,
   };

   @Module({
     imports: [
       BullModule.forRoot({
         connection: {
           host: process.env.REDIS_HOST || 'localhost',
           port: parseInt(process.env.REDIS_PORT || '6379', 10),
         },
       }),
       BullModule.registerQueue(
         { name: IMPORT_QUEUE, defaultJobOptions },
         { name: AI_GENERATION_QUEUE, defaultJobOptions }
       ),
       BullBoardModule.forRoot({
         route: '/admin/queues',
         adapter: ExpressAdapter,
       }),
       BullBoardModule.forFeature(
         { name: IMPORT_QUEUE, adapter: BullMQAdapter },
         { name: AI_GENERATION_QUEUE, adapter: BullMQAdapter }
       ),
     ],
     providers: [ImportProcessor, AiGenerationProcessor],
     exports: [BullModule],
   })
   export class QueueModule {}
   ```
